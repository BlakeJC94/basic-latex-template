% ================================================================
% SET DOCUMENT PARAMETERS

% Main
\documentclass[
  10pt,    % default font size
  a4paper  % paper size
]{article}

% Margins
\usepackage{geometry}
\geometry{
  top=2cm,
  bottom=2cm,
  left=2cm,
  right=2cm
}  

% Paragraph spacing
\setlength{\parskip}{1em}

% [USER] Review mode
%   toggles doc structure and double space
\def\draft{0}  


% ================================================================
% LOAD PREAMBLE
\input{src/preamble}


% ================================================================
% WRITE DOCUMENT
\begin{document}

  % PRINT TITLEPAGE
  \input{src/titlepage}


  % PRINT NOTES IN PREFACE IF DRAFTING
  \if\draft1
  
    % Enable roman page numbering for review notes
    \pagenumbering{roman}  

    \input{src/notes/some-notes}

    \newpage

    % Use double spacing if \draft=1
    \doublespacing 
    
    % Enable arabic page numbering for document
    \pagenumbering{arabic}  
    
  \fi


  % WRITE MAIN DOCUMENT MATTER

  \input{src/sections/a-section}

  \input{src/sections/another-section}


  % WRITE FORMATTED REFERENCES
  
  \nocite{Buzsaki2012}

  \input{src/references}


\end{document}
